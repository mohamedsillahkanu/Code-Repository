# Interactive Malaria Data Analysis Demo import pandas as pd import numpy as np import matplotlib.pyplot as plt # Create sample malaria surveillance data np.random.seed(42) dates = pd.date_range('2024-01-01', periods=12, freq='M') districts = ['Western Area', 'Freetown', 'Bo', 'Kenema', 'Makeni'] # Generate sample data data = [] for district in districts: cases = np.random.poisson(100 + np.random.randint(-30, 30), 12) for i, (date, case_count) in enumerate(zip(dates, cases)): data.append({ 'Date': date, 'District': district, 'Cases': case_count, 'Population': 50000 + np.random.randint(-10000, 10000) }) df = pd.DataFrame(data) # Calculate incidence rate per 1000 population df['Incidence_Rate'] = (df['Cases'] / df['Population']) * 1000 print("🦟 MALARIA SURVEILLANCE DATA SUMMARY") print("="*50) print(f"Total Records: {len(df)}") print(f"Districts: {', '.join(df['District'].unique())}") print(f"Date Range: {df['Date'].min().strftime('%Y-%m')} to {df['Date'].max().strftime('%Y-%m')}") print("\n📊 SUMMARY STATISTICS") print("="*50) summary = df.groupby('District').agg({ 'Cases': ['sum', 'mean', 'std'], 'Incidence_Rate': ['mean', 'max'] }).round(2) print(summary) # Create visualization plt.figure(figsize=(12, 8)) # Plot 1: Cases by District over Time plt.subplot(2, 2, 1) for district in districts: district_data = df[df['District'] == district] plt.plot(district_data['Date'], district_data['Cases'], marker='o', label=district, linewidth=2) plt.title('Malaria Cases by District Over Time') plt.xlabel('Date') plt.ylabel('Number of Cases') plt.legend() plt.xticks(rotation=45) # Plot 2: Average Incidence Rate by District plt.subplot(2, 2, 2) avg_incidence = df.groupby('District')['Incidence_Rate'].mean().sort_values(ascending=False) plt.bar(avg_incidence.index, avg_incidence.values, color=['#FF6B6B', '#4ECDC4', '#45B7D1', '#96CEB4', '#FFEAA7']) plt.title('Average Incidence Rate by District') plt.xlabel('District') plt.ylabel('Cases per 1000 Population') plt.xticks(rotation=45) # Plot 3: Monthly Trend plt.subplot(2, 2, 3) monthly_cases = df.groupby('Date')['Cases'].sum() plt.plot(monthly_cases.index, monthly_cases.values, marker='o', linewidth=3, color='#FF6B6B') plt.title('Total Monthly Cases Trend') plt.xlabel('Date') plt.ylabel('Total Cases') plt.xticks(rotation=45) # Plot 4: District Comparison plt.subplot(2, 2, 4) total_cases = df.groupby('District')['Cases'].sum().sort_values(ascending=True) plt.barh(total_cases.index, total_cases.values, color=['#FF6B6B', '#4ECDC4', '#45B7D1', '#96CEB4', '#FFEAA7']) plt.title('Total Cases by District (2024)') plt.xlabel('Total Cases') plt.tight_layout() plt.show() print("\n✅ Analysis Complete!") print("🔍 Key Insights:") print(f"• Highest burden district: {total_cases.index[-1]} ({total_cases.iloc[-1]} cases)") print(f"• Lowest burden district: {total_cases.index[0]} ({total_cases.iloc[0]} cases)") print(f"• Average monthly cases: {df['Cases'].mean():.1f}") print(f"• Peak incidence rate: {df['Incidence_Rate'].max():.2f} per 1000")
